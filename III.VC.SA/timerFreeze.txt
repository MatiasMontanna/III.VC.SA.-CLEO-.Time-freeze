SCRIPT_START
{
NOP
LVAR_INT timerAct timerDes timer_rev

timer: // El nombre del código
WAIT 0 // ejecución del código

 IF TEST_CHEAT "timer"
   IF timer_rev = 0
     PRINT_HELP_STRING "Timer freezed"
     GOSUB timerAct
   ELSE
     PRINT_HELP_STRING "Timer unfreezed"
     GOSUB timerDes
   ENDIF
ENDIF
GOTO timer

timerDes:
timer_rev = 0
FREEZE_ONSCREEN_TIMER 0
RETURN

timerAct:
timer_rev = 1
FREEZE_ONSCREEN_TIMER 1
RETURN

}
SCRIPT_END
